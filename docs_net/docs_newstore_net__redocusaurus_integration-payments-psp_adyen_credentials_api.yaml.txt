servers:

paths:
  /main:
    summary: Credentials used to authenticate Adyen payment devices and APIs.
    description: |
      Set of credentials used to authenticate Adyen payment devices and APIs.
    put:
      operationId: PutAdyenMainCredentials
      summary: Update credentials for Adyen payment devices and APIs
      description: >
        Updates the set of credentials used to authenticate Adyen payment
        devices and APIs.
      security:
        - newStoreAccessToken: []
      requestBody:
        content:
          application/json:
            schema:
              title: >-
                Updates the set of credentials used to authenticate Adyen
                payment devices and APIs.
              type: object
              description: >-
                Set of credentials used to authenticate Adyen payment devices
                and APIs.
              required:
                - company
                - username
                - password
                - account
              properties:
                company:
                  type: string
                  minLength: 1
                  description: >-
                    The company part of the User Name configured in Adyen. If
                    the Adyen User Name is `ws_12345@Company.Example` then the
                    company is `Company.Example`.
                username:
                  type: string
                  minLength: 1
                  description: >-
                    The user part of the User Name configured in Adyen. If the
                    Adyen User Name is `ws_12345@Company.Example` then the
                    username is `ws_12345`.
                password:
                  type: string
                  minLength: 1
                  description: The password configured in Adyen.
                account:
                  type: string
                  minLength: 1
                  description: The Adyen merchant account to be used as the default.
            examples:
              Put request:
                value:
                  company: Company.Example
                  username: ws_12345
                  password: password
                  account: Example_POS
      responses:
        '200':
          description: The credentials were successfully updated.
